    let x = 7;
    function tryRemoveFromArray(arr, number) {
        let returnedArray = [];
        for (let i = 0; i < arr.length; i++) {
            if (arr[i] === number) {
                continue;
            } else {
                returnedArray.push(arr[i]);
            }
        }
        return returnedArray;
    }
