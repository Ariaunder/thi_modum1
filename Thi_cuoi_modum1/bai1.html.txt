<script>
    let mark = [2, 7, 7, 3, 8, 5, 9, 10];
    let max = findMax(mark);
    function findMax(mark) {
        let max = mark[0];
        for (let i = 0; i < mark.length; i++) {
            if (mark[i] > max) {
                max = mark[i];
            }
        }
        return max;
    }
